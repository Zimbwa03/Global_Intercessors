### 🔌 Sample Code (Frontend + DeepSeek call):

```js
async function getBibleChatResponse({ phrase, chapter, verse, version }) {
  const basePrompt = `User typed: "${phrase}".`;

  const extraInfo = chapter && verse
    ? ` Give Bible verse from ${version}, Isaiah ${chapter}:${verse}, and explain it.`
    : ` Provide a Bible verse from ${version} related to "${phrase}" with a short explanation and prayer point.`;

  const response = await axios.post(
    'https://api.deepseek.com/v1/chat/completions',
    {
      model: 'deepseek-chat',
      messages: [
        {
          role: 'user',
          content: basePrompt + extraInfo,
        },
      ],
    },
    {
      headers: {
        'Authorization': 'Bearer YOUR_DEEPSEEK_API_KEY',
        'Content-Type': 'application/json',
      },
    }
  );

  return response.data.choices[0].message.content;
}
```
